64 int pivot=myArray[high];
65 int k=high;
66 int i=low;
67 int temp=0;
68 while (i<k){
69 while (myArray[i]<pivot && i<k)++i;
70 while (myArray[k]>pivot && i<k)--k;
71 if (i<k){
72 temp=myArray[i];
73 myArray[i]=myArray[k];
74 myArray[k]=temp;
75 i++;//do not decrement k here OR ELSE!!
